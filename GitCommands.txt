cd <directory>      Enter into a directory
cd ..               Get back from a directory
cd		    Resets the Directory
ls                  List of files and folders on the current directory

git init            initialize a git in the current directory
git user.username <your git username>      connect local git to a user name 
git remote add origin <Https link of a repository in github>       Adds a remote origin from github
git clone <Https link of a repository in github>        Copys all data in a repository to the current directory

git add -A    Adds the last changes to the git
git push		pushes all data from local repo to remote repo.
git commit -m "your comment"		Commits the changes to the local git
